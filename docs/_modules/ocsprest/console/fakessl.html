def cligrp(loglevel: int, verbose: int) -> None:
    """Fake CFSSL command for unit testing"""
    if verbose == 1:
        loglevel = 20
    if verbose >= 2:
        loglevel = 10
    init_logging(loglevel)
    LOGGER.setLevel(loglevel)
